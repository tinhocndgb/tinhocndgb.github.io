Viết hàm Python tính số Fibonacci thứ n bằng cả hai phương pháp: đệ quy và không đệ quy.

Đáp án Tham khảo:
```python
# Phương pháp đệ quy (đơn giản nhưng chậm với n lớn)
def fibonacci_de_quy(n):
  if n <= 1:
    return n
  return fibonacci_de_quy(n-1) + fibonacci_de_quy(n-2)

# Phương pháp không đệ quy (nhanh hơn)
def fibonacci_khong_de_quy(n):
  if n <= 1:
    return n
  a, b = 0, 1
  for _ in range(2, n + 1):
    a, b = b, a + b
  return b

# Phương pháp đệ quy với memoization (cải thiện hiệu suất)
def fibonacci_memo(n, memo={}):
  if n in memo:
    return memo[n]
  if n <= 1:
    return n
  memo[n] = fibonacci_memo(n-1, memo) + fibonacci_memo(n-2, memo)
  return memo[n]

# Ví dụ sử dụng:
n = 10
print(f"Đệ quy: F({n}) = {fibonacci_de_quy(n)}")
print(f"Không đệ quy: F({n}) = {fibonacci_khong_de_quy(n)}")
print(f"Memoization: F({n}) = {fibonacci_memo(n)}")
```

Trình bày sự khác nhau giữa phương thức append(), extend() và insert() của list trong Python. Cho ví dụ minh họa.

Đáp án Tham khảo:
- append(item):
  + Thêm một phần tử vào cuối list
  + Phần tử có thể là bất kỳ kiểu dữ liệu nào
  + Ví dụ: list.append(5) hoặc list.append([1,2])

- extend(iterable):
  + Thêm từng phần tử của iterable vào cuối list
  + "Mở rộng" list bằng cách thêm các phần tử từ iterable
  + Ví dụ: list.extend([1,2,3]) thêm 3 phần tử riêng biệt

- insert(index, item):
  + Chèn phần tử vào vị trí chỉ định
  + Các phần tử sau vị trí chèn được dịch sang phải
  + Ví dụ: list.insert(0, 'first') chèn vào đầu list

Ví dụ minh họa:
```python
my_list = [1, 2, 3]

# append()
my_list.append(4)
print(my_list) # [1, 2, 3, 4]

my_list.append([5, 6])
print(my_list) # [1, 2, 3, 4, [5, 6]]

# extend()
my_list = [1, 2, 3]
my_list.extend([4, 5])
print(my_list) # [1, 2, 3, 4, 5]

my_list.extend("ab")
print(my_list) # [1, 2, 3, 4, 5, 'a', 'b']

# insert()
my_list = [1, 2, 3]
my_list.insert(1, 99) # Chèn 99 vào vị trí index 1
print(my_list) # [1, 99, 2, 3]
```

Viết chương trình Python quản lý danh sách sản phẩm với các chức năng: thêm, xóa, tìm kiếm, hiển thị và tính tổng giá trị.

Đáp án Tham khảo:
```python
def quan_ly_san_pham():
  danh_sach_sp = []
  
  while True:
    print("\n=== QUẢN LÝ SẢN PHẨM ===")
    print("1. Thêm sản phẩm")
    print("2. Xóa sản phẩm")
    print("3. Tìm kiếm sản phẩm")
    print("4. Hiển thị tất cả sản phẩm")
    print("5. Tính tổng giá trị")
    print("6. Thoát")
    
    chon = input("Chọn chức năng (1-6): ")
    
    if chon == "1": # Thêm sản phẩm
      ma_sp = input("Nhập mã sản phẩm: ")
      ten_sp = input("Nhập tên sản phẩm: ")
      try:
        gia = float(input("Nhập giá: "))
        so_luong = int(input("Nhập số lượng: "))
      except ValueError:
        print("Giá và số lượng phải là số!")
        continue
      
      san_pham = {
        "ma": ma_sp,
        "ten": ten_sp,
        "gia": gia,
        "so_luong": so_luong
      }
      danh_sach_sp.append(san_pham)
      print("Đã thêm sản phẩm!")
    
    elif chon == "2": # Xóa sản phẩm
      ma_xoa = input("Nhập mã sản phẩm cần xóa: ")
      found = False
      for i, sp in enumerate(danh_sach_sp):
        if sp["ma"] == ma_xoa:
          danh_sach_sp.pop(i)
          print("Đã xóa sản phẩm!")
          found = True
          break
      if not found:
        print("Không tìm thấy sản phẩm!")
    
    elif chon == "3": # Tìm kiếm
      tu_khoa = input("Nhập mã hoặc tên sản phẩm: ").lower()
      ket_qua = []
      for sp in danh_sach_sp:
        if tu_khoa in sp["ma"].lower() or tu_khoa in sp["ten"].lower():
          ket_qua.append(sp)
      if ket_qua:
        print(f"Tìm thấy {len(ket_qua)} sản phẩm:")
        for sp in ket_qua:
          print(f" Mã: {sp['ma']}, Tên: {sp['ten']}, Giá: {sp['gia']}, SL: {sp['so_luong']}")
      else:
        print("Không tìm thấy sản phẩm!")
    
    elif chon == "4": # Hiển thị
      if not danh_sach_sp:
        print("Danh sách trống!")
      else:
        print(f"Có {len(danh_sach_sp)} sản phẩm:")
        for i, sp in enumerate(danh_sach_sp, 1):
          print(f"{i}. Mã: {sp['ma']}, Tên: {sp['ten']}, Giá: {sp['gia']}, SL: {sp['so_luong']}")
    
    elif chon == "5": # Tính tổng giá trị
      tong_gia_tri = sum(sp["gia"] * sp["so_luong"] for sp in danh_sach_sp)
      print(f"Tổng giá trị hàng hóa: {tong_gia_tri:,.0f} VNĐ")
    
    elif chon == "6": # Thoát
      print("Cảm ơn đã sử dụng chương trình!")
      break
    
    else:
      print("Lựa chọn không hợp lệ!")

# Chạy chương trình
quan_ly_san_pham()
```
